export const handleSidebarClick = (e, form) => {
  e.preventDefault();
  return async dispatch => {
    if (form === 'logout') {
      try {
        await axios.get('http://localhost:3000/user/logout');
        console.log('logout action called');
        dispatch({
          type: AUTH,

        });
        dispatch({
          type: USER_FORM_DISPLAY,
          display: ''
        });
      } catch (e) {
        return dispatch({
          type: ERROR,
          error: e.response.data.error
        });
      }
    } else {
      dispatch({
        type: USER_FORM_DISPLAY,
        display: form
      });
    }
  };
};
